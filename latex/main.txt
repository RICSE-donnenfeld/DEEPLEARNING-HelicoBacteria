% ---------- Abstract ----------
\begin{abstract}
\noindent
We use the Quiron Salut dataset of 245 gastric biopsies (immunohistochemically stained whole-slide images) to implement and compare two patch-level models for \textit{Helicobacter pylori} detection.
A patch-based pipeline processes $256\times256$ patches extracted along tissue borders.
Two approaches are implemented: (i) a supervised convolutional neural network (CNN) trained on expert-annotated patches, and (ii) an unsupervised autoencoder (AE) trained on healthy patches and used for anomaly detection via reconstruction error.
This report shows that both models are running and have been compared on the same validation set: we evaluate them with the comparison script (\texttt{compare\_patch\_classifiers.py}), report patch-level accuracy and precision, and include a side-by-side confusion-matrix plot.
Scope of this first report is patch-level comparison; patient-level aggregation and evaluation on the independent hold-out set are planned as next steps.
\end{abstract}

\section{Introduction}
\label{sec:intro}

Detection of \textit{Helicobacter pylori} in IHC-stained gastric biopsies is done on whole-slide images (WSI) that are too large for direct classification; we use a patch-based pipeline on $256\times256$ patches along tissue borders.
Task: (1) patch-level model (healthy vs bacteria present), (2) patient-level aggregation via contamination ratio and threshold, (3) evaluation with a held-out set.
Constraints: patient-level split to avoid leakage, limited annotated patches, class imbalance (few positive patches).
This report compares two patch-level models---supervised CNN and unsupervised autoencoder anomaly detection---on the same validation set.

\section{Methodology}
\label{sec:methodology}

\subsection{Pipeline overview}
(1) Train patch-level model (CNN or AE) on annotated patches. (2) Apply to all patches per patient. (3) Aggregate via contamination ratio $r_p$ and threshold $\tau_{patient}$.

\subsection{Preprocessing}
RGB patches resized to $256\times256$, then normalized per channel: $x' = (x - \mu)/\sigma$ with $\mu = \sigma = (0.5, 0.5, 0.5)$ (PyTorch \texttt{Normalize}), so values lie roughly in $[-1,1]$. CNN: augmentations (flips, small rotations, optional color jitter).

\subsection{Patch-level approach 1: CNN classifier}
Lightweight CNN (three conv blocks + FC). Training uses cross-entropy loss (PyTorch \texttt{CrossEntropyLoss}): for one-hot label $y$ and predicted class probabilities $\hat{p}_c$ (softmax over logits),
\[
\mathcal{L}_{CE} = -\sum_{c \in \{0,1\}} y_c \log(\hat{p}_c).
\]
Classes: $c=0$ healthy, $c=1$ unhealthy (H.\ pylori present). Trained with Adam; early stopping optional.

\subsection{Patch-level approach 2: Autoencoder anomaly detection}
AE trained on healthy patches. Training minimises mean squared error (MSE) over pixels:
\[
\mathcal{L}_{AE} = \frac{1}{|\Omega|}\|x - \hat{x}\|_2^2 = \frac{1}{CHW}\sum_{c,i,j}(x_{c,i,j} - \hat{x}_{c,i,j})^2,
\]
where $\Omega$ is the set of pixel indices (channels $C{=}3$, height $H$, width $W$). The per-patch anomaly score is either (i) the same scalar (mean over pixels), or (ii) the maximum over $8\times8$ spatial cell means (\texttt{max\_local}), so that focal anomalies drive the score. In code: \texttt{reconstruction\_error\_per\_sample} with \texttt{aggregation=\textquotedbl mean\textquotedbl} or \texttt{\textquotedbl max\_local\textquotedbl}.

Patch positive if $s(x) > \tau_{patch}$; $\tau_{patch}$ from validation ROC (point closest to (FPR=0, TPR=1), \texttt{roc\_threshold\_optimal}).

\subsection{Patient-level aggregation}
Contamination ratio per patient: $r_p = \frac{1}{N_p}\sum_{i=1}^{N_p} \hat{z}_i$, where $\hat{z}_i \in \{0,1\}$ is the patch prediction (1 if patch $i$ is predicted positive) and $N_p$ is the number of patches for patient $p$. Patient predicted positive if $r_p > \tau_{patient}$. Both $\tau_{patch}$ and $\tau_{patient}$ are fixed from cross-validation before hold-out evaluation.

\section{Experimental Design}
\label{sec:experiments}

\subsection{Train/validation split}
Splits are at \textbf{patient level} so no patient appears in both train and validation.
We use a single random split: 80\% of Cross-Validation patients for training, 20\% for validation (\texttt{val\_ratio=0.2}, \texttt{seed=42} in \texttt{model.py}, \texttt{model\_autoencoder.py}, and \texttt{compare\_patch\_classifiers.py} so the same validation set is used for comparison).
Target for full reporting is $k$-fold over patients; results in this report use one split.
The Hold-out set (116 patients) is used only once after $\tau_{patch}$ and $\tau_{patient}$ are fixed from cross-validation.

\subsection{Metrics}
Patch and patient level: confusion matrix, accuracy, precision, recall, specificity, F1, ROC-AUC.
With multiple folds we will report confidence intervals and paired tests (e.g.\ Wilcoxon) between CNN and AE.

\section{Discussion}
\label{sec:discussion}

On the validation fold (730 patches), the CNN achieves higher accuracy (0.989) and precision (0.978) than the autoencoder (0.959 and 0.940), with fewer false positives (7 vs 19) and false negatives (1 vs 11). The AE remains competitive and does not require positive patch labels for training, which is advantageous with limited annotations. Both models have high recall (CNN 0.997, AE 0.965), important for a screening setting. Limitations: single split (no CIs or p-values yet), few annotated patches, stain/scanner variability, and threshold-based aggregation vs.\ MIL. Future work: full $k$-fold with statistical tests, patient-level aggregation and hold-out evaluation, MIL for patient-level, stain normalization.

% ==========================================================
\section{Conclusion}
\label{sec:conclusion}
Two patch-level models (CNN and AE) are implemented, trained, and compared on the same validation set. On 730 validation patches the CNN reaches 0.989 accuracy and 0.978 precision; the AE reaches 0.959 and 0.940 with no supervised labels for positives. Methodology, code, confusion matrices (Figure~\ref{fig:compare_confusion}), and ROC curve (Figure~\ref{fig:roc}) are reported. Next steps: full $k$-fold with CIs and statistical tests, patient-level aggregation, and hold-out evaluation.

\begin{lstlisting}[language=Python, caption={CNN for patch classification (\texttt{model.py}).},label=lst:cnn]
class SimpleCNN(nn.Module):
    def __init__(self, num_classes: int = 2) -> None:
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 16, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2),
            nn.Conv2d(16, 32, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2),
            nn.Conv2d(32, 64, 3, padding=1), nn.ReLU(), nn.MaxPool2d(2),
        )
        self.classifier = nn.Sequential(
            nn.Flatten(), nn.Linear(64 * 32 * 32, 128), nn.ReLU(),
            nn.Linear(128, num_classes),
        )
    def forward(self, x): return self.classifier(self.features(x))
\end{lstlisting}

\begin{lstlisting}[language=Python, caption={AE per-sample reconstruction error (\texttt{model\_autoencoder.py}). \texttt{mean}: pixel MSE; \texttt{max\_local}: max over 8$\times$8 cell means.},label=lst:ae_error]
def reconstruction_error_per_sample(model, x, aggregation="mean"):
    x_hat = model(x)
    err = (x - x_hat).pow(2).mean(dim=1, keepdim=True)  # (B, 1, H, W)
    if aggregation == "mean":
        return err.view(x.size(0), -1).mean(dim=1)
    if aggregation == "max_local":
        B, _, H, W = err.shape
        cell = 8
        nc_h, nc_w = H // cell, W // cell
        err = err.view(B, 1, nc_h, cell, nc_w, cell).mean(dim=(3, 5))
        return err.view(B, -1).max(dim=1)[0]
\end{lstlisting}

\begin{lstlisting}[language=Python, caption={ROC-based patch threshold: closest to (FPR=0, TPR=1) (\texttt{roc\_threshold\_optimal}, \texttt{model\_autoencoder.py}).},label=lst:roc]
def roc_threshold_optimal(errors, labels, positive_class=1):
    order = np.argsort(-errors)
    sorted_errors, sorted_labels = errors[order], (labels[order] == positive_class).astype(np.float64)
    n_pos, n_neg = sorted_labels.sum(), len(sorted_labels) - n_pos
    tpr = np.cumsum(sorted_labels) / n_pos
    fpr = np.cumsum(1.0 - sorted_labels) / n_neg
    thresh = np.concatenate([[sorted_errors[0]+1], sorted_errors, [sorted_errors[-1]-1]])
    thresh = (thresh[:-1] + thresh[1:]) / 2
    dist = fpr**2 + (1 - tpr)**2
    idx = np.argmin(dist)
    return float(thresh[idx]), float(tpr[idx]), float(fpr[idx])
\end{lstlisting}
